\begin{frame}{Reading guide}
\small
\begin{itemize}
  \item \textbf{Colors}: treatment group versus all other couples.
  \item \textbf{Columns}: wife / husband outcomes.
  \item \textbf{Rows}: couples with youngest child age 0--10 vs.\ 11--17 at baseline.
  \item \textbf{COVID figures}: COVID-study waves.
  \item \textbf{Future figures}: annual aggregates from main-study follow-up waves.
\end{itemize}

\vspace{0.4em}
\textbf{Treatments shown in the deck}
\begin{itemize}
  \item \textbf{Main}: wife key worker (non-education), husband not key worker or education only.
  \item \textbf{Second}: husband in shutdown sector, wife not.
  \item \textbf{Appendix}: wife key worker (non-education), regardless of husband.
\end{itemize}
\end{frame}
