\end{block}

\begin{frame}{Data sources and timeline}
\begin{itemize}
  \item UKHLS main survey (waves I--O) and UKHLS COVID study (waves CA--CI).
  \item Baseline is built from 2019 interviews, prioritizing wave K, then J, then I.
  \item COVID descriptives use the COVID-study panel.
  \item Future outcomes use main-study follow-up waves J--O.
  \item In the future-outcomes branch, we restrict to interviews from March 2020 onward.
\end{itemize}
\end{frame}

\begin{frame}{Samples used in the descriptives}
\textbf{COVID sample}
\begin{itemize}
  \item Individuals in the baseline analytic sample who are observed in at least one COVID-study wave.
\end{itemize}

\vspace{0.4em}

\textbf{Couples sample}
\begin{itemize}
  \item Baseline couples defined using reciprocal partner identifiers.
  \item Qualifying relationship types:
    \begin{itemize}
      \item married,
      \item civil partnership,
      \item cohabiting / living as a couple.
    \end{itemize}
  \item Restricted to opposite-sex couples.
  \item Couple status is fixed at baseline.
\end{itemize}

\vspace{0.4em}

\textbf{Future outcomes sample}
\begin{itemize}
  \item Individuals from the baseline sample linked to main-study waves J--O.
  \item Restricted to interviews from March 2020 onward.
\end{itemize}
\end{frame}

\begin{frame}{How the baseline is constructed}
\begin{itemize}
  \item Goal: define pre-COVID characteristics that are fixed before the pandemic shock.
  \item Use the most recent interview in calendar year 2019:
    \begin{itemize}
      \item prefer wave K,
      \item then wave J,
      \item then wave I.
    \end{itemize}
  \item Baseline variables include demographics, employment, earnings, industry (SIC), occupation (SOC), work-from-home information, and family structure.
\end{itemize}
\end{frame}

\begin{frame}{Baseline group definitions used in the descriptives}
\begin{itemize}
  \item All grouping variables are based on \textbf{baseline coarse industry and occupation codes}.
  \item Shutdown sectors follow the \textbf{IFS definition}.
  \item Key-worker groups follow \textbf{clear-cut ONS-based definitions} using combinations of industry and occupation.
  \item We use two versions:
    \begin{itemize}
      \item \textbf{3-group version:} shutdown / key worker / other.
      \item \textbf{5-group version:} shutdown / health / education-childcare / other essential / other.
    \end{itemize}
  \item Appendix slides list the exact SIC and SOC conditions used in the code.
\end{itemize}
\end{frame}

\begin{frame}{Work outside home: exact descriptive rule}
\begin{itemize}
  \item \texttt{workoutside = 1} for someone who:
    \begin{itemize}
      \item is employed,
      \item has positive working hours,
      \item and is classified as not working from home.
    \end{itemize}
  \item The rule uses workplace / work-from-home information from the survey.
  \item WFH information is only directly available in the waves where the WFH questions are asked.
  \item In those waves, work outside is defined relative to the reported WFH margin.
  \item In other waves, work location is inferred from the available workplace indicators.
\end{itemize}
\end{frame}

\begin{frame}{Working from home at least sometimes}
\begin{itemize}
  \item \texttt{wfh\_some = 1} if the respondent reports working from home:
    \begin{itemize}
      \item always,
      \item often,
      \item or sometimes.
    \end{itemize}
  \item \texttt{wfh\_some = 0} if the respondent:
    \begin{itemize}
      \item never works from home,
      \item or is not working.
    \end{itemize}
  \item This is meant to capture the extensive margin of any meaningful work from home.
\end{itemize}
\end{frame}

\begin{frame}{Appendix: exact shutdown-sector definition}
\begin{itemize}
  \item Shutdown sector is defined using baseline SIC (coarse industry) codes.
  \item Exact rule:
  \[
    \texttt{base\_jbsic07\_cc} \in \{55,56,79,90,91,92,93,96\}
  \]
  \item Code meanings:
    \begin{itemize}
      \item 55 = Accommodation
      \item 56 = Food and beverage service activities
      \item 79 = Travel agency, tour operator, reservation service
      \item 90 = Creative, arts and entertainment activities
      \item 91 = Libraries, archives, museums, and other cultural activities
      \item 92 = Gambling and betting activities
      \item 93 = Sports activities and amusement / recreation activities
      \item 96 = Other personal service activities
    \end{itemize}
  \item These are the sectors treated as ``shutdown'' in the baseline grouping.
\end{itemize}
\end{frame}

\begin{frame}{Appendix: exact health key-worker definition}
\begin{itemize}
  \item Health / social-care key worker if \textbf{any} of the following hold:
    \begin{enumerate}
      \item \texttt{base\_jbsic07\_cc = 86}
      \item \texttt{base\_jbsoc10\_cc} $\in \{124,221,222,223,321\}$
      \item \texttt{base\_jbsoc10\_cc = 118} \textbf{and} \texttt{base\_jbsic07\_cc} $\in \{86,87,88\}$
      \item \texttt{base\_jbsoc10\_cc = 614} \textbf{and} \texttt{base\_jbsic07\_cc} $\in \{84,86,87,88\}$
    \end{enumerate}
  \item Main code meanings:
    \begin{itemize}
      \item SIC 86 = Human health activities
      \item SIC 87 = Residential care activities
      \item SIC 88 = Social work activities without accommodation
      \item SIC 84 = Public administration and defence; compulsory social security
      \item SOC 124 = Managers and proprietors in health and care services
      \item SOC 221 = Health professionals
      \item SOC 222 = Therapy professionals
      \item SOC 223 = Nursing and midwifery professionals
      \item SOC 321 = Health associate professionals
      \item SOC 118 = Health and social services managers / directors
      \item SOC 614 = Caring personal services
    \end{itemize}
\end{itemize}
\end{frame}

\begin{frame}{Appendix: exact education / childcare key-worker definition}
\begin{itemize}
  \item Education / childcare key worker if \textbf{any} of the following hold:
    \begin{enumerate}
      \item \texttt{base\_jbsic07\_cc = 85}
      \item \texttt{base\_jbsoc10\_cc = 231}
      \item \texttt{base\_jbsoc10\_cc} $\in \{323,612,623\}$ \textbf{and} \texttt{base\_jbsic07\_cc = 84}
    \end{enumerate}

      \item SIC 84 = Public administration and defence; compulsory social security
      \item SOC 231 = Teaching and educational professionals
      \item SOC 323 = Teaching and other educational associate professionals
      \item SOC 612 = Childcare and related personal services
      \item SOC 623 = Elementary cleaning and related occupations
    \end{itemize}
\end{itemize}
\end{frame}

\begin{frame}{Appendix: exact other-essential key-worker definition}
\begin{itemize}
  \item Other-essential key worker if \textbf{either} of the following hold:
    \begin{enumerate}
      \item \texttt{base\_jbsoc10\_cc = 331}
      \item \texttt{base\_jbsic07\_cc = 53}
    \end{enumerate}
  \item Main code meanings:
    \begin{itemize}
      \item SOC 331 = Protective service occupations
      \item SIC 53 = Postal and courier activities
    \end{itemize}
\end{itemize}
\end{frame}

\begin{frame}{Appendix: how the 3-group and 5-group variables are assigned}
\begin{itemize}
  \item All grouping is based entirely on baseline 2019 information.
  \item \textbf{3-group variable:}
    \begin{itemize}
      \item shutdown
      \item key worker
      \item other
    \end{itemize}
  \item \textbf{5-group variable:}
    \begin{itemize}
      \item shutdown
      \item key worker -- health / social care
      \item key worker -- education / childcare
      \item key worker -- other essential
      \item other
    \end{itemize}
  \item Hierarchy used in assignment:
    \begin{itemize}
      \item shutdown overrides key-worker status
      \item key-worker status overrides other
    \end{itemize}
\end{itemize}
\end{frame}
